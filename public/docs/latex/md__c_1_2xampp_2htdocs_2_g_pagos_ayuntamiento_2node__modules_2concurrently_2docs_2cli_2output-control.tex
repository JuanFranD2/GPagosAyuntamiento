It might be useful at times to make sure that the commands outputs are grouped together, while running them in parallel.~\newline
 This can be done with the {\ttfamily -\/-\/group} flag.


\begin{DoxyCode}{0}
\DoxyCodeLine{\$\ concurrently\ -\/-\/group\ "{}echo\ Hello\ there\ \&\&\ sleep\ 2\ \&\&\ echo\ 'General\ Kenobi!'"{}\ "{}echo\ hi\ Star\ Wars\ fans"{}}
\DoxyCodeLine{[0]\ Hello\ there}
\DoxyCodeLine{[0]\ General\ Kenobi!}
\DoxyCodeLine{[0]\ echo\ Hello\ there\ \&\&\ sleep\ 2\ \&\&\ echo\ 'General\ Kenobi!'\ exited\ with\ code\ 0}
\DoxyCodeLine{[1]\ hi\ Star\ Wars\ fans}
\DoxyCodeLine{[1]\ echo\ hi\ Star\ Wars\ fans\ exited\ with\ code\ 0}

\end{DoxyCode}
